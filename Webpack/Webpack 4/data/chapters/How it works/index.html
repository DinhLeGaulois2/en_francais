1: "dist" is created when ("production" mode):

      npm run build


2: "App.js", a React component
3: "index.html", used in "webpack.config.js"
(see "plugins")...
4: "index.js", the "entry" of "webpack.config.js"
5: "package.json"
6: "webpack.config.js"

With the "webpack.config.js" file:

const path = require('path');
const HtmlWebpackPlugin = require('html-webpack-plugin');

module.exports = {
    entry: './src/index.js',
    output: {
        path: path.join(__dirname, '/dist'),
        filename: 'index_bundle.js'
    },
    module: {
        rules: [
            { // for "Babel"
                test: /\.js$/,
                exclude: /node_modules/,
                use: { loader: 'babel-loader' }
            },
            { // for "SASS"
                test: /\.s(a|c)ss$/,
                use: ['style-loader', 'css-loader', 'sass-loader'],
            }
        ]
    },
    plugins: [
        new HtmlWebpackPlugin({ template: './src/index.html' })
    ]
}

If we execute with the development's mode

      npm run build

The "webpack.config.js" will be executed. From the "entry" point ("index.js" in "src/index.js"), we have (just for the structure of this app):

      "index.js" → "App.js" ("components/App.js") → "style.scss" ("src/style.scss") → "index.html" ("src/index.html") → "Web Browser"

The part that call the "index.html" ("src/index.html") in the "webpack.config.js"

plugins: [
    new HtmlWebpackPlugin({
        // use our "index.html" file and insert the "index_bundle.js" in it
        template: './src/index.html'
    })
]
